class Solution {
public:
    string addBinary(string a, string b) {
        long decA = atoi(a.c_str());
        long decB = atoi(b.c_str());
        
        long res = decA + decB;
        string resStr = to_string(res);
        
        cout << resStr << endl;
        
        int i = resStr.size()-1;
        
        while (i >= 1) {
            if (resStr[i] == '2') {
                resStr[i] = '0';
                resStr[i-1] += 1;
            } else if (resStr[i] == '3') {
                resStr[i] = '1';
                resStr[i-1] += 1;
            }
            
            --i;
        }
        
        if (resStr[0] == '2') {
            resStr[0] = '0';
            return '1' + resStr;
        } else if (resStr[0] == '3') {
            resStr[0] = '1';
            return '1' + resStr;
        }
        
        return resStr;
        
    }
};
